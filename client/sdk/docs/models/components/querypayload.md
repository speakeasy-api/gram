# QueryPayload

Payload for a generic org-scoped analytics query

## Example Usage

```typescript
import { QueryPayload } from "@gram/client/models/components/querypayload.js";

let value: QueryPayload = {
  from: new Date("2025-12-19T10:00:00Z"),
  groupBy: "department_name",
  to: new Date("2025-12-26T10:00:00Z"),
};
```
